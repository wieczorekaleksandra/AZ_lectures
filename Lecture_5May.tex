\section{Pokrycie wierzchoŇākowe i algorytm AVC}\label{sec:vc}

\subsection*{Problem}
Niech $G=(V,E)$ bńôdzie grafem nieskierowanym. \emph{Pokryciem wierzchoŇākowym}
(ang.\ \emph{vertex cover}) grafu $G$ nazywamy taki zbi√≥r $W\subseteq V$, Ňľe
\[
\forall e\in E\quad e\cap W\neq\emptyset,
\]
czyli kaŇľda krawńôdŇļ ma co najmniej jeden koniec w~$W$.

\begin{description}[leftmargin=2em]
\item[Instancja:] graf $G=(V,E)$.
\item[Problem:] znaleŇļńá pokrycie wierzchoŇākowe (VC) grafu $G$ o~minimalnej licznoŇõci.
\end{description}

Problem znajdowania minimalnego pokrycia wierzchoŇākowego jest NP-trudny, dlatego
rozwaŇľamy algorytm aproksymacyjny \textsc{AVC} (\emph{Approximate Vertex Cover}).

\subsection*{Algorytm}
\begin{algorithm}[H]
\caption{AVC($G$)}
\begin{algorithmic}[1]
\State $C\gets\emptyset$ \Comment{$C$ --- budowane rozwińÖzanie}
\State $F\gets E$ \Comment{$F$ --- zbi√≥r jeszcze nie pokrytych krawńôdzi}
\While{$F\neq\emptyset$}
  \State wybierz dowolnńÖ krawńôdŇļ $uv\in F$
  \State $C\gets C\cup\{u,v\}$
  \State usuŇĄ z~$F$ wszystkie krawńôdzie incydentne z~$u$ lub~$v$
\EndWhile
\State \Return $C$
\end{algorithmic}
\end{algorithm}

\subsection*{ZŇāoŇľonoŇõńá czasowa}
Niech $n=|V(G)|$ oraz $m=|E(G)|$.
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
\item linia 1: $O(1)$,
\item linia 2: $O(m)$,
\item pńôtlńô z~wierszy 3--6 moŇľna zaimplementowańá tak, aby dziaŇāaŇāa w~czasie $O(m)$,
\item linia 7: $O(1)$.
\end{itemize}
ŇĀńÖczna zŇāoŇľonoŇõńá czasowa \textsc{AVC} wynosi $O(m)$.

\subsection*{Wsp√≥Ňāczynnik aproksymacji}
\begin{twierdzenie}\label{tw:avc}
\textsc{AVC} jest algorytmem $2$-aproksymacyjnym.
\end{twierdzenie}

\begin{proof}
Niech $G$ bńôdzie grafem wejŇõciowym. Oznaczmy:
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
\item $c^{*}(G)$ --- najmniejszńÖ liczbńô wierzchoŇāk√≥w w~pokryciu wierzchoŇākowym $G$
(rozmiar pokrycia optymalnego),
\item $c(G)$ --- liczbńô wierzchoŇāk√≥w w~pokryciu wierzchoŇākowym znalezionym przez \textsc{AVC}.
\end{itemize}
Niech $B$ bńôdzie zbiorem krawńôdzi wybranych przez algorytm w~linii 4. Zbi√≥r $B$ jest
\emph{skojarzeniem} (czyli zbiorem parami rozŇāńÖcznych krawńôdzi): po wybraniu krawńôdzi $uv$
algorytm usuwa z~$F$ wszystkie krawńôdzie incydentne z~$u$ lub~$v$, wińôc Ňľadna kolejna
wybrana krawńôdŇļ nie dzieli koŇĄca z~$uv$.

Aby pokryńá krawńôdzie z~$B$, potrzeba co najmniej $|B|$ wierzchoŇāk√≥w (kaŇľda z~tych
parami rozŇāńÖcznych krawńôdzi wymaga osobnego wierzchoŇāka pokrycia). StńÖd
\[
c^{*}(G)\ \ge\ |B|.
\]
Z~drugiej strony kaŇľda wybrana krawńôdŇļ wnosi do $C$ dwa wierzchoŇāki, wińôc
\[
c(G)=2\,|B|.
\]
ŇĀńÖczńÖc obie nier√≥wnoŇõci,
\[
c(G)=2\,|B|\ \le\ 2\,c^{*}(G)
\qquad\Longrightarrow\qquad
\frac{c(G)}{c^{*}(G)}\ \le\ 2. \qedhere
\]
\end{proof}

\section{Problem pokrycia zbioru i algorytm zachŇāanny GSC}\label{sec:sc}

\subsection*{Problem}
Niech $X$ bńôdzie zbiorem skoŇĄczonym, $|X|=n$, a $\F\subseteq 2^{X}$ rodzinńÖ podzbior√≥w
zbioru $X$ takńÖ, Ňľe
\[
\bigcup_{S\in\F}S=X
\]
(kaŇľdy element $X$ naleŇľy do co najmniej jednego zbioru z~$\F$). Rodzinńô $\Acov\subseteq\F$
nazywamy \emph{pokryciem} zbioru $X$, jeŇõli
\[
\bigcup_{S\in\Acov}S=X.
\]

\begin{description}[leftmargin=2em]
\item[Instancja:] para $(X,\F)$, gdzie $X$ jest zbiorem skoŇĄczonym, $|X|=n$,
oraz $\F\subseteq 2^{X}$, $\bigcup_{S\in\F}S=X$.
\item[Problem:] znaleŇļńá pokrycie $\Acov\subseteq\F$ minimalizujńÖce $|\Acov|$.
\end{description}

\begin{przyklad}
Dla rodziny $\F=\{S_1,S_2,S_3,S_4,S_5,S_6\}$ pokrywajńÖcej $X$ podrodzina
$\{S_1,S_2,S_3\}$ moŇľe byńá pokryciem zbioru $X$.
\end{przyklad}

\subsection*{Algorytm}
\begin{algorithm}[H]
\caption{GSC($X,\F$) --- Greedy Set Cover}
\begin{algorithmic}[1]
\State $U\gets X$ \Comment{$U$ --- zbi√≥r jeszcze nie pokrytych element√≥w}
\State $\Acov\gets\emptyset$ \Comment{$\Acov$ --- budowane rozwińÖzanie}
\While{$U\neq\emptyset$}
  \State wybierz $S\in\F$ maksymalizujńÖcy $|S\cap U|$
  \State $\Acov\gets\Acov\cup\{S\}$
  \State $U\gets U-S$
\EndWhile
\State \Return $\Acov$
\end{algorithmic}
\end{algorithm}

\subsection*{PoprawnoŇõńá}
Pńôtla obraca sińô dop√≥ki sńÖ jeszcze nie pokryte elementy, a~w~kaŇľdym obrocie pńôtli
jakieŇõ elementy zostajńÖ pokryte (wybrany $S$ maksymalizuje $|S\cap U|$, a~poniewaŇľ
$\F$ pokrywa $X$ i~$U\neq\emptyset$, to $|S\cap U|\ge 1$). Zatem algorytm zatrzymuje
sińô i~zwraca pokrycie $X$.

\subsection*{ZŇāoŇľonoŇõńá czasowa}
\begin{itemize}[leftmargin=1.6em,itemsep=1pt]
\item linia 1: $O(n)$,
\item linia 2: $O(1)$,
\item liczba iteracji pńôtli z~wierszy 3--6 jest $\le|X|$ oraz $\le|\F|$, wińôc jest
$\le\min\{|X|,|\F|\}$; wewnńÖtrz pojedynczej iteracji:
\begin{itemize}[leftmargin=1.6em,itemsep=0pt]
\item linia 4: $O(|X|\cdot|\F|)$,
\item linia 5: $O(1)$,
\item linia 6: $O(|X|)$,
\end{itemize}
co daje $O(|X|\cdot|\F|)$ na iteracjńô,
\item linia 7: $O(1)$.
\end{itemize}
Ostatecznie zŇāoŇľonoŇõńá czasowa \textsc{GSC} wynosi
\[
O\bigl(|X|\cdot|\F|\cdot\min\{|X|,|\F|\}\bigr).
\]

\subsection*{Wsp√≥Ňāczynnik aproksymacji}
Przez $H$ oznaczamy liczby harmoniczne:
\[
H(0)=0,\qquad H(m)=1+\tfrac12+\tfrac13+\dots+\tfrac1m\quad\text{dla }m\ge 1.
\]

\begin{twierdzenie}\label{tw:gsc}
\textsc{GSC} jest algorytmem $H\!\bigl(\max\{|S|:S\in\F\}\bigr)$-aproksymacyjnym.
\end{twierdzenie}

\begin{proof}
Niech $(X,\F)$ bńôdzie instancjńÖ wejŇõciowńÖ. Oznaczmy przez $\Acov^{*}$ optymalne
pokrycie dla $(X,\F)$, a~przez $\Acov$ pokrycie znalezione przez \textsc{GSC}.
Niech $S_i$ bńôdzie zbiorem wybranym przez algorytm w~linii 4 w~$i$-tej iteracji pńôtli.

\medskip
\noindent\textbf{Przypisanie koszt√≥w.} KaŇľdemu elementowi
$x\in S_i-(S_1\cup\dots\cup S_{i-1})$ (czyli elementowi pokrytemu po raz pierwszy
w~$i$-tej iteracji) przypisujemy koszt
\[
c_x=\frac{1}{\bigl|S_i-(S_1\cup\dots\cup S_{i-1})\bigr|}.
\]
Zbi√≥r $S_i-(S_1\cup\dots\cup S_{i-1})$ to zbi√≥r element√≥w pokrytych po raz pierwszy
w~$i$-tej iteracji, wińôc caŇākowity koszt przypisany podczas jednej iteracji wynosi $1$.
PoniewaŇľ $|\Acov|$ jest liczbńÖ iteracji pńôtli, mamy
\[
|\Acov|=\sum_{x\in X}c_x.
\]

\medskip
\noindent\textbf{Oszacowanie przez pokrycie optymalne.} Skoro $\Acov^{*}$ jest pokryciem,
kaŇľdy $x\in X$ wystńôpuje w~co najmniej jednym zbiorze $S\in\Acov^{*}$, wińôc
\[
\sum_{S\in\Acov^{*}}\ \sum_{x\in S}c_x\ \ge\ \sum_{x\in X}c_x=|\Acov|.
\]
PokaŇľemy, Ňľe
\[
\sum_{x\in S}c_x\ \le\ H(|S|)\qquad\text{dla kaŇľdego }S\in\F.
\]
Niech $S\in\F$. Dla $i\in\{1,2,\dots,|\Acov|\}$ niech
\[
n_i=\bigl|S-(S_1\cup\dots\cup S_i)\bigr|
\]
(liczba element√≥w $S$, kt√≥re nie sńÖ pokryte po $i$-tej iteracji) oraz $n_0=|S|$.
Niech $k$ bńôdzie najmniejsze takie, Ňľe $n_k=0$ (wtedy $S$ jest pokryty przez
$S_1\cup S_2\cup\dots\cup S_k$). OczywiŇõcie $n_{i-1}\ge n_i$ dla $i\in[k]$, a~r√≥Ňľnica
$n_{i-1}-n_i$ to liczba element√≥w $S$ pokrytych po raz pierwszy przez $S_i$. Zatem
\[
\sum_{x\in S}c_x=\sum_{i=1}^{k}(n_{i-1}-n_i)\cdot
\frac{1}{\bigl|S_i-(S_1\cup\dots\cup S_{i-1})\bigr|}.
\]
Na podstawie wyboru zachŇāannego w~linii 4 mamy
\[
\bigl|S_i-(S_1\cup\dots\cup S_{i-1})\bigr|\ \ge\
\bigl|S-(S_1\cup\dots\cup S_{i-1})\bigr|=n_{i-1}
\qquad\text{dla kaŇľdego }S\in\F-\{S_1,\dots,S_{i-1}\}.
\]
StńÖd
\begin{align*}
\sum_{x\in S}c_x
&\le\sum_{i=1}^{k}(n_{i-1}-n_i)\cdot\frac{1}{n_{i-1}}
=\sum_{i=1}^{k}\ \sum_{j=n_i+1}^{n_{i-1}}\frac{1}{n_{i-1}}
\ \le\ \sum_{i=1}^{k}\ \sum_{j=n_i+1}^{n_{i-1}}\frac{1}{j}\\[2pt]
&=\sum_{i=1}^{k}\Bigl(\sum_{j=1}^{n_{i-1}}\frac1j-\sum_{j=1}^{n_i}\frac1j\Bigr)
=\sum_{i=1}^{k}\bigl(H(n_{i-1})-H(n_i)\bigr)\\[2pt]
&=H(n_0)-H(n_k)=H(|S|)-H(0)=H(|S|),
\end{align*}
gdzie wykorzystaliŇõmy teleskopowanie sumy.

\medskip
\noindent\textbf{ZŇāoŇľenie oszacowaŇĄ.}
\begin{align*}
|\Acov|=\sum_{x\in X}c_x
&\le\sum_{S\in\Acov^{*}}\ \sum_{x\in S}c_x
\ \le\ \sum_{S\in\Acov^{*}}H(|S|)
\ \le\ \sum_{S\in\Acov^{*}}\max\{H(|S|):S\in\Acov^{*}\}\\
&=|\Acov^{*}|\cdot\max\{H(|S|):S\in\Acov^{*}\}
\ \le\ |\Acov^{*}|\cdot\max\{H(|S|):S\in\F\}\\
&=|\Acov^{*}|\cdot H\!\bigl(\max\{|S|:S\in\F\}\bigr),
\end{align*}
a~wińôc
\[
\frac{|\Acov|}{|\Acov^{*}|}\ \le\ H\!\bigl(\max\{|S|:S\in\F\}\bigr). \qedhere
\]
\end{proof}

\begin{wniosek}\label{wn:gsc-ln}
\textsc{GSC} jest algorytmem $(\ln|X|+1)$-aproksymacyjnym.
\end{wniosek}
\begin{proof}
PoniewaŇľ $H(n)\le\ln n+1$, to
\[
H\!\bigl(\max\{|S|:S\in\F\}\bigr)\ \le\ H(|X|)\ \le\ \ln|X|+1. \qedhere
\]
\end{proof}

\subsection*{Pokrycie wierzchoŇākowe jako szczeg√≥lny przypadek pokrycia zbioru}
Pokrycie wierzchoŇākowe jest szczeg√≥lnym przypadkiem problemu pokrycia zbioru. Dla
$G=(V,E)$ przyjmujemy
\[
X=E,\qquad \F=\{E(v):v\in V\},\qquad\text{gdzie}\quad E(v)=\{e\in E: v\in e\},
\]
przy czym $|E(v)|=d(v)$ (stopieŇĄ wierzchoŇāka $v$). Wtedy
\[
\max\{|E(v)|:E(v)\in\F\}=\Delta(G),
\]
wińôc \textsc{GSC} daje dla pokrycia wierzchoŇākowego wsp√≥Ňāczynnik $H(\Delta(G))$.
PoniewaŇľ
\[
H(3)=1+\tfrac12+\tfrac13=\tfrac{11}{6}<2<\tfrac{25}{12}=H(4),
\]
dla graf√≥w o~maksymalnym stopniu $\le 3$ oszacowanie wsp√≥Ňāczynnika aproksymacyjnego
\textsc{GSC} jest lepsze niŇľ oszacowanie dla \textsc{AVC}.

\section{Schematy aproksymacyjne}\label{sec:schemes}

Niech $\Pi$ bńôdzie problemem optymalizacyjnym (NP-trudnym).

\begin{definicja}[schemat aproksymacyjny]
Algorytm $A$ jest \emph{schematem aproksymacyjnym} dla $\Pi$, jeŇõli dla dowolnego
wejŇõcia $(I,\eps)$, gdzie $I$ jest instancjńÖ $\Pi$ oraz $\eps>0$, algorytm $A$ jest
algorytmem $(1+\eps)$-aproksymacyjnym.
\end{definicja}

\begin{definicja}[wielomianowy schemat aproksymacyjny, PTAS]
Schemat aproksymacyjny $A$ nazywamy \emph{wielomianowym schematem aproksymacyjnym}
(PTAS), jeŇõli jego zŇāoŇľonoŇõńá czasowa moŇľe byńá ograniczona funkcjńÖ wielomianowńÖ
wzglńôdem $|I|$ dla kaŇľdego ustalonego $\eps>0$.
\end{definicja}
ZŇāoŇľonoŇõńá czasowa wielomianowego schematu aproksymacyjnego moŇľe wynosińá np.
$\Theta\!\bigl(2^{1/\eps}\,n^2\bigr)$, gdzie $n=|I|$. Wtedy zwińôkszanie precyzji
(czyli zmniejszanie $\eps$) jest kosztowne czasowo.

\begin{definicja}[w peŇāni wielomianowy schemat aproksymacyjny, FPTAS]
Schemat aproksymacyjny $A$ nazywamy \emph{w~peŇāni wielomianowym schematem
aproksymacyjnym} (FPTAS), jeŇõli jego zŇāoŇľonoŇõńá czasowa moŇľe byńá ograniczona funkcjńÖ
wielomianowńÖ wzglńôdem $|I|$ oraz $\tfrac1\eps$.
\end{definicja}
ZŇāoŇľonoŇõńá czasowa w~peŇāni wielomianowego schematu aproksymacyjnego moŇľe wynosińá np.
$\Theta\!\bigl(n^2\cdot\tfrac1\eps\bigr)$, gdzie $n=|I|$.
